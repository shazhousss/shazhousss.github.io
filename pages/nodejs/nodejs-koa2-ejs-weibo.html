# 安装所需框架

npm i -g koa-generator  //安装koa2脚手架
koa2 -e 项目名   //koa2初始化项目
npm install   //安装完了后，安装依赖
npm run dev   //启动项目
浏览器访问 localhost:3000
npm i cross-env -D  //安装环境变量工具
再打开package.json
    dev设置成："dev": "cross-env NODE_ENV=dev ./node_modules/.bin/nodemon bin/www",
    prd设置成："prd": "cross-env NODE_ENV=production pm2 start bin/www",
